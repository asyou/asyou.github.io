Vue项目封装WebSocket代码

2022-07-13

之前的文章写了如何在PHP和JS中创建并连接使用WebSocket进行通信，参考《Websocket实现站内即时消息提醒以及定时任务》

本文记录的是在Vue项目中封装WebSocket并使用。

一、封装代码

// /utils/websocket.js
/**
 * WebSocket 连接配置
 * @author wangwei@254555778@qq.com
 * @date 2022-07-12
 */
import store from '@/store'
const wsUrl = 'ws://socket地址/IP:端口'
let socket = null
let lockReconnect = false
let tt = null
let globalCallback = null,
    sendData = null
let createSocket = (url) => {
    try {
	if ('WebSocket' in window) {
	    socket = new WebSocket(url)
	} else if ('MozWebSocket' in window) {
	    socket = new MozWebSocket(url)
	}
	initSocket()
    } catch (e) {
	reconnet(url)
    }
}
let sendMsg = (data, callback) => {
    if (socket.readyState === 1) {
	globalCallback = callback
	socket.send(JSON.stringify(data))
    } else {
        setTimeout(() => {
	    console.log('Waiting for socket connection..')
	    sendMsg(data, callback)
        }, 1500)
	return false
    }
    socket.onmessage = (event) => {
	callback && callback(event)
    }
}
let initSocket = () => {
    socket.onopen = () => {
        console.log('Socket connection succeeded.')
	store.state.app.socket = true
	heartCheck.start() // 心跳检测
	// 可在这里做身份验证
    }

    socket.onmessage = (event) => {
	let data = JSON.parse(event)
	switch (data.type) {
	    case 'heartbeat': // 心跳检测
		break
	    case 'clues': // 线索提醒
		break
	    case 'schedule': // 日程提醒
		break
	    case 'notice': // 成交广播
		break
	    case 'system': // 系统通知
		break
	}
    }

    socket.onerror = () => {
	console.log('Socket connection failed.')
	store.state.app.socket = false
	clearInterval(heartCheck.timer1)
	clearInterval(heartCheck.timer2)
	reconnet(wsUrl)
    }

    socket.onclose = () => {
	console.log('Socket connection closed.')
	store.state.app.socket = false
	clearInterval(heartCheck.timer1)
	clearInterval(heartCheck.timer2)
	reconnet(wsUrl)
    }
}
let reconnet = (url) => {
    if (lockReconnect) {
	return false
    }
    lockReconnect = true;
    tt && clearTimeout(tt)
    tt = setTimeout(function() {
	createSocket(url)
	lockReconnect = false
    }, 5000)
}
let heartCheck = {
    timer1: null,
    timer2: null,
    start: function() {
	this.timer1 = setInterval(function() {
	    socket.send(JSON.stringify({
		type: 'heartbeat',
		msg: 'ping'
	    }))
	}, 30000)
	this.timer2 = setInterval(function() {
	    console.log('check schedule')
	}, 60000)
    }
}
createSocket(wsUrl)
export default {
    sendMsg
}

二、全局引入

// /main.js
import socket from '@/utils/websocket.js' // 引用
Vue.prototype.socket = socket // 挂载到Vue原型链上

三、其他页面中使用

// /index.vue
onload() {
    this.socket.sendMsg({
	type: 'msg',
	uuid: '用户ID',
	data: '消息内容'
    }, (event) => {
	console.log('authorize')
    })
}

因为socket已挂载到全局，所以可以直接使用this.socket
